\begin{algorithm}
\KwOut{satisfiability of the given CNF}
\SetKwComment{Comment}{}{}
\BlankLine
\Begin{
\While{there is an unassigned or unpropagated var}{
  \If{there is no unpropagated var}{
    select a decision var and its phase\;
  }
  \If{BCP() returns an conflict}{
    \If{decision level is root level}{
      \Return{UNSAT}\;
    }
    CONFLICT-ANALYSIS()\;
    BACKTRACK()\;
    \uIf{there are too many learnt clauses}{
      RESTART()\;
      REDUCE()\;
      ELIMINATE()\;
      VIVIFY()\;
      update restart threshold and interval\Comment*{\sl\small\color[gray]{0.5}// Luby Stabilization}
      select var phases;
    }
    \ElseIf{LBD-based dynamic restart condition holds}{
      RESTART();
    }
  }
}
\Return{SAT}\;
}
